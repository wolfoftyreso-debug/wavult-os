Data Import Guide

Migrate existing data into pixdrift via the API. This guide covers bulk import of contacts, companies, deals, and historical ledger entries.

Rate limits: Bulk imports hit rate limits. Use a delay between requests or contact support for a temporary limit increase during migration.

Import from CSV

A complete Node.js script to import contacts from a CSV file:

// import-contacts.js
const fs = require('fs')
const csv = require('csv-parse/sync')

const API_KEY = process.env.PIXDRIFT_API_KEY
const BASE = 'https://api.bc.pixdrift.com'

async function importContacts(csvFile) {
  const content = fs.readFileSync(csvFile, 'utf8')
  const records = csv.parse(content, {
    columns: true,
    skip_empty_lines: true
  })

  console.log(`📥 Importing ${records.length} contacts...`)
  
  const results = { success: 0, failed: 0, errors: [] }

  for (const [i, row] of records.entries()) {
    try {
      const res = await fetch(`${BASE}/api/contacts`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${API_KEY}`,
          'Content-Type': 'application/json'
        },
        body: JSON.stringify({
          first_name: row.first_name || row.FirstName,
          last_name: row.last_name || row.LastName,
          email: row.email || row.Email,
          phone: row.phone || row.Phone,
          title: row.title || row.Title
        })
      })

      if (res.ok) {
        results.success++
      } else {
        const err = await res.json()
        results.failed++
        results.errors.push({ row: i + 1, error: err.error })
      }
    } catch (err) {
      results.failed++
      results.errors.push({ row: i + 1, error: err.message })
    }

    // Rate limit: max 100 req/15min = ~6.7/min → 1 per 200ms is safe
    if (i > 0 && i % 50 === 0) {
      console.log(`  Progress: ${i}/${records.length}`)
      await sleep(2000) // Extra pause every 50 requests
    } else {
      await sleep(200)
    }
  }

  console.log(`✅ Done: ${results.success} imported, ${results.failed} failed`)
  if (results.errors.length > 0) {
    console.log('Errors:', results.errors)
  }
}

const sleep = ms => new Promise(r => setTimeout(r, ms))

// Usage: node import-contacts.js contacts.csv
importContacts(process.argv[2])

Import companies + contacts (linked)

async function importWithCompanies(data) {
  // data = [{ company: {...}, contacts: [...] }]

  for (const { company, contacts } of data) {
    // 1. Create company first
    const co = await post('/api/companies', company)
    console.log(`✅ Company: ${co.name} (${co.id})`)

    // 2. Create contacts linked to company
    for (const contact of contacts) {
      const c = await post('/api/contacts', { ...contact, company_id: co.id })
      console.log(`  → Contact: ${c.first_name} ${c.last_name}`)
      await sleep(200)
    }
    await sleep(500)
  }
}

async function post(path, body) {
  const res = await fetch(BASE + path, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${API_KEY}`,
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body)
  })
  if (!res.ok) throw new Error(`${path} failed: ${await res.text()}`)
  return res.json()
}

Import historical ledger entries

Accounting note: Each ledger entry must balance (debits = credits). Verify your source data before importing. Consider importing one month at a time. 
async function importLedgerEntries(entries) {
  // entries = [{ description, reference, date, lines: [{account_id, debit, credit}] }]
  
  for (const entry of entries) {
    // Validate balance before sending
    const totalDebit = entry.lines.reduce((s, l) => s + (l.debit || 0), 0)
    const totalCredit = entry.lines.reduce((s, l) => s + (l.credit || 0), 0)
    
    if (Math.abs(totalDebit - totalCredit) > 0.001) {
      console.error(`⚠️ Unbalanced entry: ${entry.reference} (${totalDebit} ≠ ${totalCredit})`)
      continue
    }

    const result = await post('/api/ledger/entries', entry)
    console.log(`✅ Entry: ${entry.reference} (${result.id})`)
    await sleep(300)
  }
}

Import deals from HubSpot / Salesforce

// Example: transform HubSpot deal format to pixdrift
function transformHubSpotDeal(hs) {
  return {
    title: hs.properties.dealname,
    value: parseFloat(hs.properties.amount || '0'),
    currency: hs.properties.deal_currency_code || 'SEK',
    status: mapHubSpotStage(hs.properties.dealstage),
    expected_close_date: hs.properties.closedate?.split('T')[0]
  }
}

function mapHubSpotStage(stage) {
  const map = {
    appointmentscheduled: 'prospecting',
    qualifiedtobuy: 'qualification',
    presentationscheduled: 'proposal',
    decisionmakerboughtin: 'negotiation',
    contractsent: 'negotiation',
    closedwon: 'closed_won',
    closedlost: 'closed_lost'
  }
  return map[stage] || 'prospecting'
}

// Fetch from HubSpot API and import
const hubspotDeals = /* ... fetch from HubSpot ... */
for (const hsDeal of hubspotDeals) {
  await post('/api/deals', transformHubSpotDeal(hsDeal))
  await sleep(200)
}

Import checklist

Before import:
  ✅ Test with 5-10 records first
  ✅ Validate CSV encoding (UTF-8)
  ✅ Check required fields per endpoint
  ✅ Prepare a rollback plan (note what you imported)

During import:
  ✅ Log all created IDs with original source IDs
  ✅ Respect rate limits (sleep between requests)
  ✅ Handle errors gracefully (don't abort on first failure)

After import:
  ✅ Verify counts: contacts, companies, deals
  ✅ Spot-check 5 records manually
  ✅ Run GET /api/dashboards/admin to see totals
Need to import large datasets (10,000+ records)? Contact api@pixdrift.com for a temporary rate limit increase during your migration window.
