Webhooks Integration Guide

A complete guide to building webhook receivers for pixdrift events using Node.js / Express.

Complete Express webhook handler

A production-ready webhook handler with signature verification, idempotency, and error handling:

// webhook-server.js
const express = require('express')
const crypto = require('crypto')

const app = express()
const WEBHOOK_SECRET = process.env.PIXDRIFT_WEBHOOK_SECRET

// Track processed event IDs to prevent duplicate processing
const processedEvents = new Set()

// IMPORTANT: Must use raw body for HMAC verification — NOT express.json()
app.post('/webhooks/pixdrift', express.raw({ type: '*/*' }), async (req, res) => {
  // 1. Verify signature
  const signature = req.headers['x-pixdrift-signature']
  if (!verifySignature(req.body, signature, WEBHOOK_SECRET)) {
    console.error('⛔ Invalid webhook signature')
    return res.status(401).json({ error: 'Invalid signature' })
  }

  const event = JSON.parse(req.body.toString())

  // 2. Idempotency check
  if (processedEvents.has(event.id)) {
    console.log(`⏭️ Duplicate event: ${event.id}`)
    return res.status(200).json({ received: true, duplicate: true })
  }
  processedEvents.add(event.id)

  // 3. Acknowledge immediately (async processing)
  res.status(200).json({ received: true })

  // 4. Process event asynchronously
  handleEvent(event).catch(err => {
    console.error(`❌ Event processing error [${event.id}]:`, err)
  })
})

function verifySignature(payload, signature, secret) {
  if (!signature || !secret) return false
  const expected = 'sha256=' + crypto
    .createHmac('sha256', secret)
    .update(payload)
    .digest('hex')
  try {
    return crypto.timingSafeEqual(Buffer.from(expected), Buffer.from(signature))
  } catch {
    return false
  }
}

async function handleEvent(event) {
  console.log(`📨 Event: ${event.type} [${event.id}]`)

  switch (event.type) {
    // ── Deals ───────────────────────────────────
    case 'deal.won': {
      const { title, value, currency } = event.data
      await notifySlack(
        `🎉 Deal won: *${title}* — ${value.toLocaleString()} ${currency}`
      )
      await updateCRM(event.data)
      break
    }
    case 'deal.created':
      await syncToHubSpot(event.data)
      break

    // ── Tasks ────────────────────────────────────
    case 'task.overdue':
      await sendEmailReminder(event.data.assigned_to, event.data)
      break
    case 'task.completed':
      await logActivity(event.data)
      break

    // ── Non-conformance ──────────────────────────
    case 'non_conformance.opened':
      await alertQualityTeam(event.data)
      break
    case 'non_conformance.closed':
      await updateQualityDashboard(event.data)
      break

    default:
      console.log(`Unhandled event type: ${event.type}`)
  }
}

// Example handler implementations
async function notifySlack(message) {
  if (!process.env.SLACK_WEBHOOK_URL) return
  await fetch(process.env.SLACK_WEBHOOK_URL, {
    method: 'POST',
    body: JSON.stringify({ text: message }),
    headers: { 'Content-Type': 'application/json' }
  })
}

app.listen(3000, () => console.log('🪝 Webhook server running on port 3000'))

Retry handling

pixdrift retries failed webhooks with exponential backoff: immediately, then after 5 minutes, then after 30 minutes.

Critical: Always return 200 within 30 seconds. Use a queue (BullMQ, SQS, etc.) to handle slow processing. If your handler throws, pixdrift will retry — potentially causing duplicate side effects.

Production checklist

✅ Verify HMAC-SHA256 signature on every request
✅ Use raw body parser (before JSON parse) for verification
✅ Return 200 immediately — process asynchronously
✅ Implement idempotency (track event IDs in DB/cache)
✅ Use Redis/DynamoDB for processedEvents in production (not in-memory Set)
✅ Log all events with their IDs for debugging
✅ Monitor failed webhook deliveries in pixdrift admin
