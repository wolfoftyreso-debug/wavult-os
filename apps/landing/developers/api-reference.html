pixdrift API Reference

Complete reference for all API endpoints. Base URL: https://api.bc.pixdrift.com


All endpoints require Authorization: Bearer <token>. See Authentication for details.

GET
/api/contacts
List all contacts for the authenticated organization, ordered by creation date.
curl
Node.js
Python
curl -X GET https://api.bc.pixdrift.com/api/contacts \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json"
const response = await fetch('https://api.bc.pixdrift.com/api/contacts', {
  headers: { 'Authorization': 'Bearer YOUR_API_KEY' }
});
const contacts = await response.json();
import requests
response = requests.get(
  'https://api.bc.pixdrift.com/api/contacts',
  headers={'Authorization': 'Bearer YOUR_API_KEY'}
)
contacts = response.json()
Response 200
[
  {
    "id": "uuid",
    "org_id": "uuid",
    "first_name": "Anna",
    "last_name": "Svensson",
    "email": "anna@example.com",
    "phone": "+46701234567",
    "company_id": "uuid",
    "title": "CTO",
    "created_at": "2026-03-21T10:00:00Z"
  }
]

Try it

POST
/api/contacts
Create a new contact in your organization.

Request Body

FieldTypeDescription
first_name requiredstringContact's first name
last_name requiredstringContact's last name
emailstringEmail address
phonestringPhone number
company_iduuidAssociated company
titlestringJob title
curl
Node.js
Python
curl -X POST https://api.bc.pixdrift.com/api/contacts \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"first_name":"Anna","last_name":"Svensson","email":"anna@example.com"}'
const response = await fetch('https://api.bc.pixdrift.com/api/contacts', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    first_name: 'Anna',
    last_name: 'Svensson',
    email: 'anna@example.com'
  })
});
response = requests.post(
  'https://api.bc.pixdrift.com/api/contacts',
  headers={'Authorization': 'Bearer YOUR_API_KEY'},
  json={'first_name': 'Anna', 'last_name': 'Svensson'}
)

PATCH
/api/deals/{id}/status
Update deal pipeline stage. Valid values: prospecting, qualification, proposal, negotiation, closed_won, closed_lost
curl
Node.js
Python
curl -X PATCH https://api.bc.pixdrift.com/api/deals/DEAL_ID/status \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"status":"closed_won"}'
await fetch(`https://api.bc.pixdrift.com/api/deals/${dealId}/status`, {
  method: 'PATCH',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({ status: 'closed_won' })
});
requests.patch(
  f'https://api.bc.pixdrift.com/api/deals/{deal_id}/status',
  headers={'Authorization': 'Bearer YOUR_API_KEY'},
  json={'status': 'closed_won'}
)

POST
/api/ledger/entries
Create a double-entry ledger transaction. Debits and credits must balance (±0.001 tolerance).
curl
Node.js
Python
curl -X POST https://api.bc.pixdrift.com/api/ledger/entries \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "description": "Invoice payment",
    "reference": "INV-2026-001",
    "lines": [
      {"account_id": "1510", "debit": 12500, "credit": 0},
      {"account_id": "3000", "debit": 0, "credit": 12500}
    ]
  }'
await fetch('https://api.bc.pixdrift.com/api/ledger/entries', {
  method: 'POST',
  headers: {
    'Authorization': 'Bearer YOUR_API_KEY',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    description: 'Invoice payment',
    reference: 'INV-2026-001',
    lines: [
      { account_id: '1510', debit: 12500, credit: 0 },
      { account_id: '3000', debit: 0, credit: 12500 }
    ]
  })
});
requests.post(
  'https://api.bc.pixdrift.com/api/ledger/entries',
  headers={'Authorization': 'Bearer YOUR_API_KEY'},
  json={
    'description': 'Invoice payment',
    'lines': [
      {'account_id': '1510', 'debit': 12500, 'credit': 0},
      {'account_id': '3000', 'debit': 0, 'credit': 12500}
    ]
  }
)

/api/convert?amount=1000&from=USD&to=SEK
Convert an amount between currencies using the latest stored exchange rates.
curl
Node.js
Python
curl "https://api.bc.pixdrift.com/api/convert?amount=1000&from=USD&to=SEK" \
  -H "Authorization: Bearer YOUR_API_KEY"
const params = new URLSearchParams({ amount: '1000', from: 'USD', to: 'SEK' });
const result = await fetch(
  `https://api.bc.pixdrift.com/api/convert?${params}`,
  { headers: { 'Authorization': 'Bearer YOUR_API_KEY' } }
).then(r => r.json());
// { from: 'USD', to: 'SEK', amount: 1000, converted: 10342, rate: 10.342 }
result = requests.get(
  'https://api.bc.pixdrift.com/api/convert',
  params={'amount': 1000, 'from': 'USD', 'to': 'SEK'},
  headers={'Authorization': 'Bearer YOUR_API_KEY'}
).json()

POST
/api/stripe/create-checkout-session
Create a Stripe checkout session. Returns a redirect URL to Stripe hosted checkout.
curl
Node.js
Python
curl -X POST https://api.bc.pixdrift.com/api/stripe/create-checkout-session \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{"plan_id":"growth","success_url":"https://app.pixdrift.com/success","cancel_url":"https://app.pixdrift.com/billing"}'
const { url } = await fetch(
  'https://api.bc.pixdrift.com/api/stripe/create-checkout-session',
  {
    method: 'POST',
    headers: {
      'Authorization': 'Bearer YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      plan_id: 'growth',
      success_url: 'https://app.pixdrift.com/success',
      cancel_url: 'https://app.pixdrift.com/billing'
    })
  }
).then(r => r.json());
window.location.href = url; // Redirect to Stripe
session = requests.post(
  'https://api.bc.pixdrift.com/api/stripe/create-checkout-session',
  headers={'Authorization': 'Bearer YOUR_API_KEY'},
  json={
    'plan_id': 'growth',
    'success_url': 'https://app.pixdrift.com/success',
    'cancel_url': 'https://app.pixdrift.com/billing'
  }
).json()
# Redirect to session['url']
